	/**
	 * 将航班号进行转换用于比较
	 * @param a 待转换航班号
	 * @return 转换后的航班号
	 */
	public static String changeformat(String a) {  //转换航班号进行判断
		if(a.length()==6) {
			return a;
		}
		else if(a.length()==4) {
			return a.substring(0, 2)+"00"+a.substring(2);	
		}
		else{
			return a.substring(0, 2)+"0"+a.substring(2);
		}
	}
